SurfaceArea,Population
,LifeExpectancy,
GNP , round(Population/SurfaceArea,2) as pop_denc,

## country with highest percapita income
select name,Continent,Population,GNP ,round(GNP*1000000/Population) as GNP_percap from country
order by GNP_percap desc limit 1;


##Limiting the data to top
select 
code,name,
Continent,Region,
SurfaceArea,Population
,LifeExpectancy,
GNP , round(Population/SurfaceArea,2) as pop_denc from country

name,
Continent,
Population
,
GNP , round(GNP*1000000/Population , 2) as per_capita_inc from country

##display the data of countries within rank 11 to 15 as per  per_cap_inc

-- offset = ignore rows
-- limit = top n no of  rows
select 
name,
Continent,
Population
,
GNP , round(GNP*1000000/Population , 2) as per_capita_inc from country
order by per_capita_inc desc
limit 5
offset 10;

select

Continent,
Population
,
GNP , round(GNP*1000000/Population , 2) as per_capita_inc from country

 ##logical operators - And/Or
 
select Name,Continent,IndepYear,SurfaceArea,Population,LifeExpectancy,GNP from country
where  Continent= "Asia" order by Name desc;

##display the data of all asian countries where population > 10 m
select Name,Continent,IndepYear,SurfaceArea,Population,LifeExpectancy,GNP from country
where Continent = "Asia" And  Population > 10000000 order by Name desc;
-- and means both condition must be satisfied

select Name,Continent,IndepYear,SurfaceArea,Population,LifeExpectancy,GNP from country
where Continent = "Asia" or  Population > 10000000 order by Name desc;
-- or means any one  condition must be satisfied

#  display the data of  al countries Asia,Europe,Africa

select Name,Continent,IndepYear,SurfaceArea,Population,LifeExpectancy,GNP from country
where Continent = "Asia" or Continent="Europe" or Continent="Africa"
